\newcommand*{\fakeminipage}[1]{%
\begin{minipage}[b]{0pt}%
\setlength{\vfuzz}{\stockwidth}%
\setlength{\hfuzz}{\stockheight}%
#1\end{minipage}%
\setlength{\vfuzz}{0pt}%
\setlength{\hfuzz}{0pt}%
}
